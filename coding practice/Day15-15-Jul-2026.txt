Day 15 - Automation with Terraform, Ansible, CI/CD Pipelines
🔹 Ansible — Deploy a Web Application to Multiple Hosts
- name: Deploy web application to multiple hosts							→ Gives a name to the Ansible playbook.
  hosts: all												→ Runs the playbook on all servers in the inventory.
  become: yes												→ Executes tasks with root (sudo) privileges.
  tasks:												→ Starts the list of automation tasks.
    - name: Copy web app files										→ Names the task for copying application files.
      copy:												→ Copies files from the Ansible control node to target servers.
        src: /path/to/web/app/										→ Specifies the source directory on the control node.
        dest: /var/www/html/										→ Specifies the destination directory on the target server.
    - name: Start web app service									→ Names the task for starting the web server.
      service:												→ Manages Linux services.
        name: apache2											→ Specifies the Apache web server service.
        state: started											→ Starts the Apache service.
        enabled: yes											→ Starts Apache automatically after every reboot.


🔹 Terraform — Create EC2 with ALB (Application Load Balancer)
resource "aws_lb" "vishwesh_lb" {									→ Creates an AWS Application Load Balancer.
  name               = "vishwesh-lb"									→ Sets the Load Balancer name.
  internal           = false										→ Creates a public internet-facing Load Balancer.
  load_balancer_type = "application"									→ Creates an Application Load Balancer (ALB).
  security_groups    = [aws_security_group.allow_ssh_http.id]						→ Attaches the specified Security Group.
  subnets            = [aws_subnet.public.id]								→ Deploys the ALB in the specified public
}

resource "aws_instance" "vishwesh_ec2" {								→ Creates an EC2 instance.
  ami             = "ami-0abcd1234abcd1234"								→ Specifies the Amazon Machine Image (AMI).
  instance_type   = "t2.micro"										→ Launches a t2.micro EC2 instance.
  security_groups = [aws_security_group.allow_ssh_http.name]						→ Attaches the Security Group to the EC2 instance.

  tags = {												→ Assigns the EC2 instance name tag.
    Name = "vishwesh-EC2"
  }
}
🔹 Kubernetes — Helm Chart for Redis with Persistence
helm install vishwesh-redis bitnami/redis --set persistence.enabled=true --set persistence.size=8Gi	→ Installs the Bitnami Redis Helm chart. [--set persistence.enabled=true] → Enables persistent storage for Redis. [--set persistence.size=8Gi] → Allocates an 8 GiB Persistent Volume for Redis data.
🔹 Jenkinsfile — Multi-Environment Deployment
pipeline {												→ Starts the Jenkins pipeline.
    agent any												→ Runs the pipeline on any available Jenkins agent.
    environment {											→ Defines environment variables for the pipeline.
        IMAGE_NAME = 'vishwesh-web-app'									→ Sets the Docker image name.
        DEPLOY_ENV = 'production'									→ Sets the deployment environment.
    }
    stages {												→ Starts the pipeline stages.
        stage('Build') {										→ Defines the build stage.
            steps {
                echo 'Building Docker image...'								→ Prints the build message.
                docker.build("${env.IMAGE_NAME}")							→ Builds the Docker image using the environment variable.
            }
        }
        stage('Test') {											→ Defines the testing stage.
            steps {											
                echo 'Running tests...'									→ Prints the testing message.
            }
        }
        stage('Deploy to Kubernetes') {									→ Defines the deployment stage.
            steps {
                script {										→ Runs conditional Groovy script.
                    if (env.DEPLOY_ENV == 'production') {						→ Checks whether the deployment target is production.
                        sh 'kubectl apply -f kubernetes/prod-deployment.yaml'				→ Deploys the production Kubernetes manifest.
                    } else {										→ Executes when the environment is not production.
                        sh 'kubectl apply -f kubernetes/dev-deployment.yaml'				→ Deploys the development Kubernetes manifest.
                    }
                }
            }
        }
    }
}
🔹 GitLab CI/CD — Environment-Specific Deployments
stages:													→ Defines the CI/CD pipeline stages.
  - build												→ First stage builds the application.
  - push												→ Second stage pushes the Docker image.
  - deploy												→ Third stage deploys the application.

build:													→ Starts the build job.
  stage: build
  script:
    - docker build -t vishwesh-web-app .								→ Builds the Docker image.

push:													→ Starts the image push job.
  stage: push
  script:
    - docker login -u "$CI_REGISTRY_USER" -p "$CI_REGISTRY_PASSWORD"					→ Logs in to the Docker registry.
    - docker push vishwesh-web-app									→ Pushes the Docker image to the registry

deploy:													→ Starts the deployment job.
  stage: deploy
  script:
    - |
      if [ "$CI_COMMIT_REF_NAME" == "main" ]; then							→ Checks whether the current branch is main.
        kubectl apply -f kubernetes/prod-deployment.yaml
      else												Executes for all other branches.
        kubectl apply -f kubernetes/dev-deployment.yaml							→ Deploys to the development environment.
      fi


🔹 Docker
# Multi RUN Commands
FROM ubuntu												→ Uses Ubuntu as the base image.
RUN apt update && \											→ Updates the package repository.
    apt install curl -y && \										→ Installs the Curl package.									
    apt install vim -y											→ Installs the Vim editor.
CMD ["echo", "Hello vishwesh"]										→ Executes echo "Hello vishwesh" when the container starts.
